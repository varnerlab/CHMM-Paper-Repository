\subsection{Assumptions}
\label{sec:assumptions}

We use four assumptions throughout. They are mild and standard in the
HMM literature, but stating them explicitly closes a gap in the v10
manuscript, where the existence of a unique stationary distribution,
the compactness of the parameter space, and the identifiability of
the emission family were all used implicitly.

\begin{assumption}[Irreducibility and aperiodicity]
\label{ass:irred}
The transition matrix $\mathbf T \in \R^{K \times K}$ is irreducible
and aperiodic on the state space $\{1, \ldots, K\}$.
\end{assumption}

\begin{assumption}[Finite second moment and emission contrast]
\label{ass:moments}
Each per-state emission density $b_k(\cdot;\boldsymbol\theta_k)$ has
finite second moment, $\E_{b_k}[G^2] < \infty$, and at least two
states differ in either location $\mu_k$ or scale $\sigma_k$ (or
$b_k$ for the Laplace family).
\end{assumption}

\begin{assumption}[Compact parameter space]
\label{ass:compact}
The parameter space $\Theta$ is compact in the topology used by EM:
$\mu_k \in [-M, M]$ for some $M < \infty$, $\sigma_k \in [\sigma_{\min}, \sigma_{\max}]$
or $b_k \in [b_{\min}, b_{\max}]$ with $\sigma_{\min}, b_{\min} \ge 10^{-6}$,
and for CHMM-t the per-state degrees of freedom satisfy
$\nu_k \in [\nu_{\min}, \nu_{\max}]$ with $\nu_{\min} \ge 2.1$ and
$\nu_{\max} < \infty$. Transition entries lie in the open simplex
interior with $T_{ij} \ge \epsilon_T > 0$ for some $\epsilon_T$.
\end{assumption}

\begin{assumption}[True parameter inside the parameter space]
\label{ass:dgp}
The observed return series $G_{1:T}$ is generated by a single fixed
parameter $\theta_0 \in \mathrm{int}(\Theta)$.
Assumption~\ref{ass:dgp} is used only for the consistency statement
in Proposition~\ref{prop:consistency}; the other results in this
section do not require it.
\end{assumption}

Under Assumption~\ref{ass:irred} the chain $\mathbf T$ has a unique
stationary distribution $\bar{\boldsymbol\pi}$ satisfying
$\bar{\boldsymbol\pi}\,\mathbf T = \bar{\boldsymbol\pi}$, and there
exists a spectral decomposition with $\lambda_1 = 1$ a simple eigenvalue
and all other eigenvalues $\lambda_2, \ldots, \lambda_K$ strictly inside
the unit disk; the chain is geometrically ergodic at the rate
$\rho(\mathbf T) = \max_{k \ge 2} |\lambda_k| < 1$
\citep{levinperes2017markov}. Compactness in
Assumption~\ref{ass:compact} (in particular $\nu_k \le \nu_{\max} < \infty$)
prevents the Student-t emission from collapsing to a Gaussian and the
Gaussian or Laplace from collapsing to a point mass, so identifiability
inside the parameter space is intact.

\subsection{Spectral mechanism for the absolute-return ACF}
\label{sec:acf_mechanism}

Let $G_t$ be a stationary CHMM, $\bar{\boldsymbol\pi}$ the stationary
distribution of $\mathbf T$, and define the per-state absolute-return
moments
\begin{equation}
    m_k \;=\; \E\!\left[\,|G_t|\;\big|\; s_t = k\right],
    \qquad
    M_k \;=\; \E\!\left[\,G_t^2\;\big|\; s_t = k\right],
    \qquad k = 1, \ldots, K.
    \label{eq:per_state_moments}
\end{equation}
For the Gaussian emission $b_k = \mathcal N(\mu_k, \sigma_k^2)$ these
are $m_k = \sigma_k\sqrt{2/\pi}\,\exp(-\mu_k^2/2\sigma_k^2)
+ \mu_k(1 - 2\Phi(-\mu_k/\sigma_k))$ and $M_k = \mu_k^2 + \sigma_k^2$;
for the Student-t and Laplace families the corresponding closed forms
are standard.

\begin{theorem}[Mixture-of-eigenvalues form for the absolute-return ACF]
\label{thm:acf_mixture}
Under Assumptions~\ref{ass:irred}-\ref{ass:moments}, for every
$\tau \ge 1$ the absolute-return autocovariance of a stationary CHMM
satisfies
\begin{equation}
    \mathrm{Cov}\!\left(|G_t|,\, |G_{t+\tau}|\right)
    \;=\; \sum_{k=2}^{K} c_k\, \lambda_k^{\tau},
    \label{eq:acf_exact}
\end{equation}
with $\lambda_2, \ldots, \lambda_K$ the non-unit eigenvalues of
$\mathbf T$ and the coefficients $c_k$ given by
\begin{equation}
    c_k \;=\; \big(\mathbf m^\top \mathrm{diag}(\bar{\boldsymbol\pi})\,
                  \mathbf v_k\big)\,
              \big(\mathbf w_k^\top \mathbf m\big),
    \qquad k = 2, \ldots, K,
    \label{eq:acf_coefficients}
\end{equation}
where $\mathbf m = (m_1, \ldots, m_K)^\top$ and $(\mathbf v_k, \mathbf w_k)$
are the right and left eigenvectors of $\mathbf T$ associated with
$\lambda_k$, normalised so that $\mathbf w_k^\top \mathbf v_k = 1$
and $\mathbf w_1 = \bar{\boldsymbol\pi}$, $\mathbf v_1 = \mathbf 1$.
The autocorrelation function is
\begin{equation}
    \rho_{|G|}(\tau)
    \;=\; \frac{1}{\sigma_{|G|}^2}\sum_{k=2}^{K} c_k\, \lambda_k^{\tau}
    \;=\; \sum_{k=2}^{K} w_k\, \lambda_k^{\tau},
    \qquad
    w_k = c_k / \sigma_{|G|}^2,
    \label{eq:acf_normalised}
\end{equation}
with $\sigma_{|G|}^2 = \sum_k \bar{\boldsymbol\pi}_k M_k - (\sum_k \bar{\boldsymbol\pi}_k m_k)^2 - \E[G_t]^2 + (\E|G_t|)^2$
the variance of $|G_t|$.
\end{theorem}

\begin{proof}
Conditioning on the latent state pair $(s_t, s_{t+\tau})$ and using
that emissions are conditionally independent given the state sequence,
\begin{equation*}
    \E\!\left[|G_t|\,|G_{t+\tau}|\right]
    = \sum_{i,j} \Pr(s_t = i,\, s_{t+\tau} = j)\,
                 \E\!\left[|G_t|\;\big|\; s_t = i\right]
                 \E\!\left[|G_{t+\tau}|\;\big|\; s_{t+\tau} = j\right]
    = \sum_{i,j} \bar{\boldsymbol\pi}_i\, (\mathbf T^\tau)_{ij}\, m_i\, m_j
    = \mathbf m^\top \mathrm{diag}(\bar{\boldsymbol\pi})\, \mathbf T^\tau\, \mathbf m.
\end{equation*}
Under Assumption~\ref{ass:irred} the spectral decomposition of
$\mathbf T$ reads
$\mathbf T^\tau = \mathbf 1\bar{\boldsymbol\pi}^\top
+ \sum_{k=2}^K \lambda_k^\tau\, \mathbf v_k \mathbf w_k^\top$,
so
\begin{equation*}
    \mathbf m^\top \mathrm{diag}(\bar{\boldsymbol\pi})\, \mathbf T^\tau\, \mathbf m
    = \mathbf m^\top \bar{\boldsymbol\pi}\, \bar{\boldsymbol\pi}^\top \mathbf m
    + \sum_{k=2}^K \lambda_k^\tau\,
                   \mathbf m^\top \mathrm{diag}(\bar{\boldsymbol\pi})\, \mathbf v_k\,
                   \mathbf w_k^\top \mathbf m.
\end{equation*}
The first term is $(\bar{\boldsymbol\pi}^\top \mathbf m)^2 = (\E|G_t|)^2$,
and subtracting $\E|G_t|\,\E|G_{t+\tau}| = (\E|G_t|)^2$ yields
the autocovariance in equation~\eqref{eq:acf_exact} with $c_k$ as in
equation~\eqref{eq:acf_coefficients}. Dividing by $\sigma_{|G|}^2$
gives the autocorrelation in equation~\eqref{eq:acf_normalised}.
\end{proof}

\begin{remark}
\label{rem:acf_mixture_squared}
The same derivation with $|G_t|$ replaced by $G_t^2$ and $\mathbf m$
replaced by $(M_1, \ldots, M_K)^\top$ gives the analogous spectral
form for the squared-return ACF; the proof is identical line-by-line.
\end{remark}

Theorem~\ref{thm:acf_mixture} is the precise version of the
mixture-of-exponentials approximation $\rho_{|G|}(\tau) \approx
\sum_k w_k \lambda_k^\tau$ used informally in the
discussion of Section~\ref{sec:discussion} (equation~\eqref{eq:acf_mixture}).
The approximation is in fact an exact equality at the population level;
the empirical ACF deviates from this form only through the standard
Monte Carlo and finite-sample noise of the simulator.

\begin{corollary}[Ryd\'{e}n separation as a $K$-rank statement]
\label{cor:ryden_separation}
At $K = 2$, equation~\eqref{eq:acf_normalised} reduces to the
mono-exponential form $\rho_{|G|}(\tau) = w_2\, \lambda_2^\tau$, and
matching a target slow-decay rate $\rho^*(\tau) = \exp(-\tau/\tau^*)$
on a horizon $\tau \in \{1, \ldots, L\}$ with $\tau^* \ge L$ forces
$|\lambda_2| \ge \exp(-1/\tau^*)$, i.e.\ a near-unit second eigenvalue.
A Markov chain on $K = 2$ states with $|\lambda_2| \to 1$ has a
diagonal-dominant transition matrix in which both states are nearly
absorbing; under Assumption~\ref{ass:moments} this collapses the
stationary mixture marginal to one of the two emission densities and
fails distributional KS at the moderate horizons we evaluate.
At $K \ge 3$ the right-hand side of equation~\eqref{eq:acf_normalised}
admits at least two non-unit eigenvalues, and the ACF can match
$\rho^*(\tau)$ over the same horizon with each individual $|\lambda_k|$
bounded away from $1$ by a $K$-dependent margin, breaking the
distributional-versus-temporal tension.
\end{corollary}

\begin{proof}[Proof sketch]
The $K = 2$ reduction is direct from equation~\eqref{eq:acf_normalised}.
For the diagonal-dominant collapse, observe that the eigenvalues of a
$2 \times 2$ stochastic matrix with diagonal entries $T_{11}, T_{22}$
are $\lambda_1 = 1$ and $\lambda_2 = T_{11} + T_{22} - 1$, so
$|\lambda_2| \to 1$ requires $T_{11} + T_{22} \to 2$. Under
Assumption~\ref{ass:moments} this concentrates the stationary
$\bar{\boldsymbol\pi}$ near a vertex of the simplex and the marginal
mixture $\bar{\boldsymbol\pi}_1 b_1 + \bar{\boldsymbol\pi}_2 b_2$
collapses to a single emission density with the wrong tail. The
$K \ge 3$ case follows because the linear system
$\sum_{k=2}^K w_k \lambda_k^\tau = \rho^*(\tau)$, $\tau = 1, \ldots, L$,
has $L$ equations and $2(K-1)$ unknowns, and the non-uniqueness allows
a solution with each $|\lambda_k|$ bounded by a $K$-dependent constant
strictly less than $1$.
\end{proof}

The corollary is a one-paragraph theoretical restatement of the
empirical Ryd\'{e}n et al.\ result: at $K = 2$ the ACF and the
marginal are coupled through a single eigenvalue, so the model cannot
match both; at $K \ge 3$ the two requirements decouple. This is the
spectral mechanism behind the moderate-$K$ recommendation in
Section~\ref{sec:k_selection} and the empirical observation that
ACF-MAE is essentially flat across $K \in \{3, 6, \ldots, 21\}$ while
KS pass rates rise.

\subsection{ECM monotonicity for CHMM-t}
\label{sec:ecm_monotonicity}

The CHMM-t M-step is an ECM \citep{mengrubin1993ecm}, which generalises
EM by replacing a single joint M-step with a sequence of
conditional-maximisation (CM) steps. The classical EM monotonicity
guarantee \citep{wu1983convergence, baum1970maximization} extends to
ECM provided each CM step does not decrease the augmented $Q$-function
\citep{mengrubin1993ecm}. We pin this property down for the specific
ECM used in CHMM-t.

\begin{proposition}[ECM monotonicity for CHMM-t]
\label{prop:ecm_monotone}
Let $Q(\theta\mid\theta^{(n)}) = \E_{S_{1:T} \mid O_{1:T}, \theta^{(n)}}\!\left[\log p(O_{1:T}, S_{1:T} \mid \theta)\right]$
be the standard EM auxiliary function for the CHMM-t with
$\theta = (\mathbf T, \boldsymbol\pi, \{(\mu_k, \sigma_k, \nu_k)\}_{k=1}^K)$.
Define the two-block CM step:
\begin{align*}
&\textbf{CM-1.}\quad
(\mathbf T^{(n+1)},\, \boldsymbol\pi^{(n+1)},\, \{(\mu_k^{(n+1)}, \sigma_k^{(n+1)})\}_{k=1}^K)
\;\leftarrow\; \arg\max_{(\mathbf T, \boldsymbol\pi, \{\mu_k, \sigma_k\})}
Q(\theta \mid \theta^{(n)})\\
&\hspace{8em}\text{with } \nu_k = \nu_k^{(n)} \text{ held fixed for all } k,\\[2pt]
&\textbf{CM-2.}\quad
\nu_k^{(n+1)} \;\leftarrow\; \arg\max_{\nu \in [\nu_{\min}, \nu_{\max}]}
Q\!\big(\theta \mid \theta^{(n)}\big)
\quad\text{with the CM-1 outputs held fixed},
\quad k = 1, \ldots, K.
\end{align*}
Under Assumption~\ref{ass:compact} the CM-1 outputs are the closed-form
weighted updates of equations~\eqref{eq:tprecision_supp} and~\eqref{eq:tupdate_supp},
and CM-2 admits a maximiser inside the bracket because $Q$ is continuous
in $\nu$ on the compact set $[\nu_{\min}, \nu_{\max}]$. Both CM steps
satisfy
\begin{equation*}
    Q(\theta^{(n+1)} \mid \theta^{(n)})
    \;\ge\; Q(\theta^{(n)} \mid \theta^{(n)}),
\end{equation*}
and consequently
\begin{equation*}
    \mathcal L(\theta^{(n+1)}) \;\ge\; \mathcal L(\theta^{(n)}),
\end{equation*}
where $\mathcal L(\theta) = \log p(O_{1:T} \mid \theta)$ is the
incomplete-data log-likelihood. The sequence $\{\mathcal L(\theta^{(n)})\}_{n \ge 1}$
is therefore non-decreasing and, being bounded above on the compact
parameter space of Assumption~\ref{ass:compact}, converges to a finite
limit.
\end{proposition}

\begin{proof}
CM-1 maximises $Q$ over the $(\mathbf T, \boldsymbol\pi, \{\mu_k, \sigma_k\})$
block, so the post-CM-1 value of $Q$ is at least the pre-CM-1 value;
the closed-form expressions follow from setting the gradient of the
weighted Student-t log-likelihood to zero with $\nu_k$ fixed. CM-2
maximises $Q$ over $\nu_k \in [\nu_{\min}, \nu_{\max}]$, so the
post-CM-2 value of $Q$ is at least the post-CM-1 value. Compactness
of $[\nu_{\min}, \nu_{\max}]$ and continuity of
$Q_k(\nu) = \sum_t \gamma_t(k)\,\log t_\nu(O_t; \mu_k, \sigma_k)$ in
$\nu$ guarantee a maximiser; the golden-section search of
Algorithm~\ref{alg:chmm_em} converges to it because $Q_k$ is
strictly unimodal on the bracket for each fixed $(\mu_k, \sigma_k)$.
The standard EM inequality
$\mathcal L(\theta^{(n+1)}) - \mathcal L(\theta^{(n)})
\;\ge\;
Q(\theta^{(n+1)} \mid \theta^{(n)}) - Q(\theta^{(n)} \mid \theta^{(n)})$
\citep{wu1983convergence, mengrubin1993ecm} converts the $Q$-monotonicity
into incomplete-data log-likelihood monotonicity. Boundedness above
on $\Theta$ follows from Assumption~\ref{ass:compact}, so the bounded
monotone sequence converges.
\end{proof}

Proposition~\ref{prop:ecm_monotone} is the formal version of the
in-text claim "ECM preserves the monotonicity of EM" used in
Section~\ref{sec:mstep} and Section~\ref{sec:baum_welch_supp}, and it
makes explicit the role of the compact bracket $[\nu_{\min}, \nu_{\max}]$.
It does not assert convergence to the global MLE; this is the standard
EM caveat \citep{wu1983convergence} and applies equally to CHMM-N and
CHMM-L.

\subsection{Identifiability and MLE consistency}
\label{sec:identifiability}

\begin{proposition}[Identifiability up to label permutation]
\label{prop:identifiability}
Under Assumptions~\ref{ass:irred}-\ref{ass:moments}, the CHMM
parameter $\theta = (\mathbf T, \boldsymbol\theta_{1:K}, \boldsymbol\pi)$
is identifiable up to label permutation in each of the three emission
families considered (CHMM-N, CHMM-t with $\nu_k \in [\nu_{\min}, \nu_{\max}]$,
CHMM-L). That is, if two parameters $\theta, \theta'$ induce the same
distribution over $G_{1:T}$ for every $T \ge 1$, then $\theta'$ is
obtained from $\theta$ by a permutation of state labels.
\end{proposition}

\begin{proof}[Proof sketch]
The general HMM identifiability theorem of \citet{allman2009identifiability}
(Theorem~6) shows that a $K$-state HMM with conditionally
non-degenerate emissions is identifiable up to label permutation
provided (i) $\mathbf T$ has full rank $K$ and (ii) the per-state
emission densities $\{b_k\}_{k=1}^K$ are linearly independent in the
space of probability densities on $\R$. Under
Assumption~\ref{ass:irred} a $K$-state irreducible aperiodic Markov
chain has a transition matrix of full rank \citep{levinperes2017markov}.
For the emission families considered:
(a) Gaussian emissions are linearly independent whenever the parameter
pairs $(\mu_k, \sigma_k)$ are distinct, by the classical mixture
identifiability of \citet{yakowitzspragins1968}.
(b) Student-t emissions with bounded $\nu_k \in [\nu_{\min}, \nu_{\max}]$
form a three-parameter exponential-tilted family for which the same
linear-independence argument applies under
Assumption~\ref{ass:moments} (distinct $(\mu_k, \sigma_k)$ across
states); the upper bracket $\nu_{\max} < \infty$ prevents the
collapse to a Gaussian and the lower bracket $\nu_{\min} \ge 2.1$
prevents the divergence of the second moment.
(c) Laplace emissions with distinct $(\mu_k, b_k)$ are linearly
independent by an identical argument.
The combination of full-rank $\mathbf T$ and linearly independent
emissions yields identifiability up to label permutation by the
\citet{allman2009identifiability} theorem.
\end{proof}

\begin{proposition}[MLE consistency, by reference]
\label{prop:consistency}
Under Assumptions~\ref{ass:irred}-\ref{ass:dgp} the maximum-likelihood
estimator $\hat\theta_T = \arg\max_{\theta \in \Theta} \mathcal L_T(\theta)$
of the CHMM is consistent: $\hat\theta_T \xrightarrow{p} \theta_0$ as
$T \to \infty$, modulo label permutation.
\end{proposition}

\begin{proof}[Proof reference]
The general HMM-MLE consistency theorem of
\citet{bickel1998asymptotic} (Theorem~1) and its extension to
non-stationary chain initialisations by
\citet{doucmoulinesryden2004} apply directly under
Assumptions~\ref{ass:irred}-\ref{ass:dgp}: irreducibility and
aperiodicity (\ref{ass:irred}) provide the required mixing condition,
finite second moment (\ref{ass:moments}) provides the moment condition,
compactness (\ref{ass:compact}) provides the parameter-space
compactness condition, and Assumption~\ref{ass:dgp} guarantees the
true parameter lies in the interior of $\Theta$ so that consistency
is meaningful. We do not reproduce the proof; the reference establishes
the result under exactly the assumption set we list.
\end{proof}

Proposition~\ref{prop:consistency} is a reference statement rather
than a new result; its function in the paper is to certify that the
MLE used implicitly in Section~\ref{sec:em_scaffold} converges to the
true parameter, without which the empirical pass-rate numbers would
be at risk of an unspecified bias. We do not pursue finite-sample
rates: at $T_{\text{IS}} = 2{,}516$ the consistency argument carries
the inferential load, and finite-sample rates for HMM-MLE would
require log-Sobolev-type controls outside the scope of this paper.

\subsection{Rank reordering preserves marginals}
\label{sec:rank_marginals}

The cross-asset construction in Section~\ref{sec:cross_asset_methods}
relies on the rank-reordering of \citet{iman1982distribution} to
inject a copula on top of the per-asset CHMM marginals without
distorting any marginal. We give a self-contained proof of the
exact-marginal property and a finite-$T$ Kolmogorov bound for the
imposed copula.

\begin{proposition}[Marginal preservation under rank reordering]
\label{prop:rank_marginals}
Let $\tilde g_{j, 1:T}$ denote a sample of size $T$ from the per-asset
CHMM marginal of asset $j$, and let $\tilde g_{j, (1)} \le \cdots \le
\tilde g_{j, (T)}$ be its order statistics. Let
$\mathbf u^{(p)} \in [0,1]^{T \times d}$ be a copula sample as in
Algorithm~\ref{alg:copula_sim}, with $r_{j,t}^{(p)} = \mathrm{rank}(u_{j,t}^{(p)})$
and the rank-reordered output
$\hat g_{j,t}^{(p)} = \tilde g_{j, (r_{j,t}^{(p)})}^{(p)}$ for $t = 1, \ldots, T$.
Then for each asset $j$, the empirical CDF of
$\{\hat g_{j,t}^{(p)}\}_{t=1}^T$ equals the empirical CDF of
$\{\tilde g_{j,t}^{(p)}\}_{t=1}^T$ exactly:
\begin{equation*}
    \hat F_j^{\hat g}(x)
    \;=\; \frac{1}{T}\sum_{t=1}^T \mathbf 1\!\left\{\hat g_{j,t}^{(p)} \le x\right\}
    \;=\; \frac{1}{T}\sum_{t=1}^T \mathbf 1\!\left\{\tilde g_{j,t}^{(p)} \le x\right\}
    \;=\; \hat F_j^{\tilde g}(x), \qquad \forall x \in \R.
\end{equation*}
\end{proposition}

\begin{proof}
The map $t \mapsto r_{j,t}^{(p)}$ is a permutation of $\{1, \ldots, T\}$
(since the copula sample $\{u_{j,t}^{(p)}\}$ has distinct values almost
surely under the continuous copula construction). Therefore the
multiset $\{\hat g_{j,t}^{(p)}\}_{t=1}^T = \{\tilde g_{j, (r_{j,t}^{(p)})}^{(p)}\}_{t=1}^T$
equals the multiset of order statistics
$\{\tilde g_{j, (k)}^{(p)}\}_{k=1}^T$, which in turn equals the
unsorted sample $\{\tilde g_{j,t}^{(p)}\}_{t=1}^T$ as a multiset.
The empirical CDF depends only on the multiset of values, so the two
empirical CDFs coincide.
\end{proof}

\begin{remark}[Copula imposed at finite $T$]
\label{rem:copula_finite_T}
The rank-reordered output has joint dependence determined by the
ranks $\{r_{j,t}^{(p)}\}$, which are an empirical copula sample. The
Kolmogorov-Smirnov distance between the empirical copula of the
rank-reordered path and the target $\mathcal C(\Sigma)$ is
$O_p(T^{-1/2})$ by the Glivenko-Cantelli theorem applied to multivariate
empirical processes \citep{iman1982distribution}; at $T = 2{,}516$ this
is below $0.02$ in our setup. The exact-marginal claim of
Proposition~\ref{prop:rank_marginals} holds at every $T$ regardless of
this finite-$T$ copula deviation.
\end{remark}

Proposition~\ref{prop:rank_marginals} is the formal version of the
"reordering only permutes simulated values, so every per-asset
marginal is preserved exactly" sentence in
Section~\ref{sec:cross_asset_methods}. It is the structural reason
the copula-versus-SIM ordering on cross-asset KS in Pipeline~B
($97.5\%$/$96.2\%$ vs.\ $72.1\%$) is mechanically anchored rather
than a sampling artefact: SIM convolves the market factor with
residuals and distorts the marginal whenever $\beta_j \ne 1$, while
rank reordering preserves it by construction.

\subsection{Filter-VaR over-conservatism}
\label{sec:filter_var_theory}

The filter-based regime-conditional VaR rule of
Section~\ref{par:filter_conditional_var} (equation~\eqref{eq:filter_conditional_var})
fails Kupiec on the $2024$-$2026$ OoS window for all three CHMM
families, with breach rates $0.2\%$ at $\alpha = 0.01$ and $1.2$-$1.4\%$
at $\alpha = 0.05$, well below the targets. We give a clean inequality
that explains the over-conservatism and pinpoints the parameter region
in which the inequality is informative.

\begin{proposition}[Mixture-quantile lower bound]
\label{prop:filter_var}
Let $\boldsymbol\pi \in [0,1]^K$ be a probability vector over the
$K$ states, and let $F_1, \ldots, F_K$ be the per-state emission CDFs.
Define the mixture CDF $F(x) = \sum_k \pi_k F_k(x)$ and the
mixture-quantile $\mathrm{VaR}^{\mathrm{mix}}(\alpha) = F^{-1}(\alpha)$.
Let $k^* = \arg\min_k F_k^{-1}(\alpha)$ denote the state with the
most extreme $\alpha$-quantile, and let $q^* = F_{k^*}^{-1}(\alpha)$.
Then
\begin{equation}
    F(q^*) \;\le\; \alpha,
    \label{eq:filter_var_inequality}
\end{equation}
with strict inequality whenever there exists $k \ne k^*$ with
$\pi_k > 0$ and $F_k(q^*) < \alpha$. As a consequence,
\begin{equation*}
    \mathrm{VaR}^{\mathrm{mix}}(\alpha) \;\le\; \min_k F_k^{-1}(\alpha),
\end{equation*}
with strict inequality under the same condition.
\end{proposition}

\begin{proof}
At $q^*$, the per-state CDF of state $k^*$ evaluates to $\alpha$ by
definition. For every $k \ne k^*$, the inequality $q^* \le F_k^{-1}(\alpha)$
implies $F_k(q^*) \le \alpha$ by monotonicity of $F_k$. Therefore
$F(q^*) = \sum_k \pi_k F_k(q^*) \le \sum_k \pi_k \alpha = \alpha$,
with strict inequality whenever the precondition holds. Since $F$ is
non-decreasing, $F(q^*) \le \alpha$ implies
$\mathrm{VaR}^{\mathrm{mix}}(\alpha) = F^{-1}(\alpha) \le q^*$;
strict inequality at the CDF level lifts to strict inequality at the
quantile level when $F$ is strictly increasing in a neighbourhood of
$q^*$.
\end{proof}

The inequality direction is empirically counter-intuitive (one might
guess that mixing produces a less extreme quantile, not a more extreme
one), so the proposition is not vacuous: it identifies the operational
mechanism behind the filter-VaR Kupiec failure. The parameter region
in which the inequality bites is exactly the region the empirical
fit lands in: the filter posterior $\boldsymbol\pi_t$ places non-trivial
mass on at least one heavy-tailed tail-state (corresponding to a
small $\nu_k$ for CHMM-t, or an outlying $(\mu_k, \sigma_k)$ for
CHMM-N and CHMM-L), and the bulk states have $F_k(q^*) \ll \alpha$
because the heavy-tailed quantile $q^*$ lies well inside the bulk
state's body. The CHMM-t penalised ECM with $\lambda = 20$ moves
$\nu_{\min}$ from $2.1$ to $4.89$ but does not eliminate the
mixing-with-bulk effect, which is why the rate sweep
$\lambda \in \{0, 5, 20, 50, 100, 200\}$ in
Section~\ref{sec:conditional_var} leaves the filter-VaR breach rate
essentially unchanged.

\begin{remark}[Remedies indicated by the inequality]
The inequality $F(q^*) \le \alpha$ with strict inequality identifies
two intervention points. (i) A concentration constraint on the filter
posterior, which forces $\pi_k \approx 1$ for the most likely
state $k^\dagger$ (so that the inequality reduces to
$F^{-1}(\alpha) = F_{k^\dagger}^{-1}(\alpha)$); this is the
operational analog of using the Viterbi-decoded state under a
sufficient-statistic restriction. (ii) An explicit state-conditional
variance overlay that replaces the heavy-tailed $F_{k^*}$ in the
mixture with a smoother target, which moves $q^*$ closer to the
bulk-state quantile and suppresses the strict inequality. Both are
listed as candidate remedies in Section~\ref{sec:conditional_var}
and are anchored in the inequality of equation~\eqref{eq:filter_var_inequality}.
\end{remark}

\subsection{Discrete-quantization VaR floor}
\label{sec:discrete_var_floor}

The discrete-HMM baselines of \citet{alswaidan2026hybrid} fail Kupiec
at $5\%$ VaR with $\text{LR}_{\text{uc}} = 16.81$, and the Poisson-jump
augmentation does not remedy the failure. We give a structural
inequality that explains why the failure is intrinsic to the
bin-centroid scaffold and not a hyperparameter artefact.

\begin{proposition}[Bin-centroid VaR floor]
\label{prop:discrete_var_floor}
Let $\{c_1 < c_2 < \cdots < c_J\}$ be the bin centroids of a
discrete-state HMM, and let
$\widehat{\mathrm{VaR}}^{\mathrm{disc}}_T(\alpha)$ denote the
$\alpha$-quantile of a simulated path of length $T$ from the
discrete-HMM scaffold; by construction
$\widehat{\mathrm{VaR}}^{\mathrm{disc}}_T(\alpha) \in \{c_1, \ldots, c_J\}$.
Let $\mathrm{VaR}^{\mathrm{cts}}(\alpha)$ denote the true continuous-population
$\alpha$-quantile of the data-generating process, and let
$j(\alpha)$ index the bin containing $\mathrm{VaR}^{\mathrm{cts}}(\alpha)$,
i.e.\ $c_{j(\alpha)} \le \mathrm{VaR}^{\mathrm{cts}}(\alpha) < c_{j(\alpha)+1}$.
Define the local bin width $\Delta_{j(\alpha)} = c_{j(\alpha)+1} - c_{j(\alpha)}$.
Then for every $T$ and every realisation,
\begin{equation*}
    \big|\widehat{\mathrm{VaR}}^{\mathrm{disc}}_T(\alpha) - \mathrm{VaR}^{\mathrm{cts}}(\alpha)\big|
    \;\ge\; \min\!\big(
       \mathrm{VaR}^{\mathrm{cts}}(\alpha) - c_{j(\alpha)},\,
       c_{j(\alpha)+1} - \mathrm{VaR}^{\mathrm{cts}}(\alpha)\big),
\end{equation*}
and the worst-case absolute error is bounded below by
$\Delta_{j(\alpha)} / 2$ when $\mathrm{VaR}^{\mathrm{cts}}(\alpha)$
lies at the bin midpoint. Adding a Poisson-jump duration mechanism
that triggers tail-state residence with intensity $\epsilon \to 0$ and
mean duration $\lambda$ does not alter the centroid set $\{c_1, \ldots, c_J\}$,
so the inequality continues to hold with the same right-hand side.
\end{proposition}

\begin{proof}
By construction the discrete-HMM simulator returns bin centroids,
so $\widehat{\mathrm{VaR}}^{\mathrm{disc}}_T(\alpha) \in \{c_j\}$.
For any centroid $c_j$, the absolute distance to
$\mathrm{VaR}^{\mathrm{cts}}(\alpha) \in [c_{j(\alpha)}, c_{j(\alpha)+1})$
is at least the smaller of the two endpoint distances, which is
maximised when $\mathrm{VaR}^{\mathrm{cts}}(\alpha)$ sits at the bin
midpoint, giving the floor $\Delta_{j(\alpha)}/2$. The Poisson-jump
augmentation only redistributes residence probabilities across the
existing centroid set, so it cannot move a centroid to the
$\mathrm{VaR}^{\mathrm{cts}}(\alpha)$ location.
\end{proof}

In Section~\ref{sec:utility} the discrete NJ and WJ baselines produce
an OoS $5\%$ VaR breach rate of $1.7\%$ against the $5\%$ target;
Proposition~\ref{prop:discrete_var_floor} explains the structural
origin: the bin-centroid quantile estimator cannot resolve a
$\mathrm{VaR}^{\mathrm{cts}}(\alpha)$ that does not coincide with a
centroid, and at $K_{\text{disc}} = 90$ the quantile bias is the
dominant source of the Kupiec rejection. Adding the prior-paper
Poisson-jump mechanism with $\epsilon = 5 \times 10^{-5}$, $\lambda = 67$
does not alter the centroid set; it only changes which centroid is
visited, which does not address the floor inequality.

\subsection{Out-of-scope theory}
\label{sec:theory_out_of_scope}

Three classes of result are deliberately not pursued in this section.
A finite-sample bound on $\hat\theta_T - \theta_0$ would require
log-Sobolev-type controls outside scope; at $T = 2{,}516$ consistency
(Proposition~\ref{prop:consistency}) carries the inferential load.
A formal proof that the weighted-Laplace MLE is the global maximiser
of the weighted Laplace likelihood is in \citet{rabiner1986introduction}
and is not reproduced here. A lower bound on the mixing time of
$\mathbf T$ as a function of $K$ would sharpen
Corollary~\ref{cor:ryden_separation} into a quantitative $K$-rank
result; this is the natural follow-up theory direction and is left
to future work.
